<script>
const history=[];let busy=false;
const SYSTEM=`You are Navigator AI — a premium autonomous task agent that handles real-world digital tasks for users with precision and elegance. When a user gives you a task: 1. Acknowledge confidently 2. Give a numbered step-by-step action plan 3. Specify exact websites/apps for each step 4. List any info needed from the user 5. Mention eligibility checks where relevant 6. Always confirm before payment or irreversible action. Use real Indian websites: MakeMyTrip, IRCTC, Rapido, pmkisan.gov.in, DigiLocker etc.`;
function quickSend(t){document.getElementById('input').value=t;sendMessage();}
function addMsg(role,text){
  const m=document.getElementById('messages');
  const d=document.createElement('div');
  d.className='msg '+role;
  d.innerHTML=`<div class="av ${role}">${role==='ai'?'N':'U'}</div><div class="bubble">${text.replace(/\n/g,'<br>').replace(/\*\*(.*?)\*\*/g,'<b>$1</b>')}</div>`;
  m.appendChild(d);m.scrollTop=m.scrollHeight;
}
function setTyping(on){
  document.getElementById('typingRow').style.display=on?'block':'none';
  document.getElementById('sendBtn').disabled=on;
  busy=on;
  document.getElementById('messages').scrollTop=9999;
}
async function sendMessage(){
  const inp=document.getElementById('input');
  const text=inp.value.trim();
  if(!text||busy)return;
  inp.value='';inp.style.height='auto';
  addMsg('user',text);
  history.push({role:'user',parts:[{text}]});
  setTyping(true);
  try{
    const r=await fetch('/api/chat',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({history,system:SYSTEM})});
    const data=await r.json();
    const reply=data.reply||'Unable to process that request.';
    history.push({role:'model',parts:[{text:reply}]});
    setTyping(false);addMsg('ai',reply);
  }catch(e){
    setTyping(false);addMsg('ai','Connection error. Please try again.');
  }
}
</script>
